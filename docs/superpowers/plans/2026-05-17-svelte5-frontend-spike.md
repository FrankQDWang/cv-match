> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a parallel `apps/web-svelte` SvelteKit spike that proves typed Workbench API access, auth/session flow, the interactive strategy graph, graph-candidate loading, and lazy safe resume snapshots before committing to a full React replacement.

**Architecture:** Keep the existing React app unchanged. - [ ] **Step 3: Add API error type**

Create `src/lib/api/errors.ts`:

```ts
export class ApiError extends Error {
  readonly status: number;

  constructor(message: string, status: number) {
    super(message);
    this.name = 'ApiError';
    this.status = status;
  }
}

export function safeErrorMessage(error: unknown, fallback: string) {
  if (!(error instanceof ApiError)) {
    return fallback;
  }
  if (error.status === 401) {
    return '请先登录。';
  }
  if (error.status === 403) {
    return '没有权限执行此操作。';
  }
  if (error.status >= 500) {
    return '服务暂时不可用，请稍后重试。';
  }
  return fallback;
}
```

Do not display `ApiError.message` directly in UI. It may contain backend `detail` text, artifact paths, provider payload markers, or header names. Only this safe helper may format user-visible API errors.

- [ ] **Step 4: Add CSRF-aware OpenAPI client**

Create `src/lib/api/client.ts`:

```ts
import createClient from 'openapi-fetch';

import { ApiError } from './errors';
import type { paths } from './schema';

const CSRF_HEADER = 'X-CSRF-Token';

function isMutating(method: string) {
  const upper = method.toUpperCase();
  return upper !== 'GET' && upper !== 'HEAD';
}

function errorMessage(error: unknown, status: number) {
  if (typeof error === 'string') {
    return error;
  }
  if (error && typeof error === 'object') {
    const payload = error as { detail?: unknown; error?: unknown; message?: unknown };
    if (typeof payload.detail === 'string') {
      return payload.detail;
    }
    if (typeof payload.error === 'string') {
      return payload.error;
    }
    if (typeof payload.message === 'string') {
      return payload.message;
    }
  }
  return `Request failed with status ${status}`;
}

export function createApiClient() {
  let csrfToken = '';

  async function credentialedFetch(input: RequestInfo | URL, init: RequestInit = {}, csrfRetry = true): Promise<Response> {
    const sourceRequest = input instanceof Request ? input : new Request(input, init);
    const request = sourceRequest.clone();
    const method = init.method ?? request.method ?? 'GET';
    const headers = new Headers(request.headers);
    for (const [key, value] of new Headers(init.headers)) {
      headers.set(key, value);
    }
    const shouldSendCsrf = csrfToken && isMutating(method) && !headers.has(CSRF_HEADER);

    if (shouldSendCsrf) {
      headers.set(CSRF_HEADER, csrfToken);
    }

    const response = await fetch(request, {
      method,
      headers,
      credentials: 'include',
    });

    const refreshedCsrf = response.headers.get(CSRF_HEADER);
    if (refreshedCsrf) {
      csrfToken = refreshedCsrf;
    }

    if (response.status === 403 && csrfRetry && shouldSendCsrf) {
      await credentialedFetch('/api/auth/me', {}, false);
      return credentialedFetch(sourceRequest.clone(), init, false);
    }

    return response;
  }

  const client = createClient<paths>({
    baseUrl: '',
    fetch: credentialedFetch,
  });

  return { client };
}

export function requireData<T>(result: { data?: T; error?: unknown; response: Response }): T {
  if (result.data !== undefined) {
    return result.data;
  }
  throw new ApiError(errorMessage(result.error, result.response.status), result.response.status);
}
```

Do not manually set `Content-Type` in this custom fetch wrapper. `openapi-fetch` should own JSON serialization and multipart/FormData boundary behavior; the wrapper only adds credentials and CSRF headers.

- [ ] **Step 5: Add Workbench API functions**

Create `src/lib/api/workbench.ts` with typed functions for the spike:

```ts
import { createApiClient, requireData } from './client';
import type { components } from './schema';

type WorkbenchEvent = components['schemas']['WorkbenchEventResponse'];

export const api = createApiClient().client;

export async function bootstrapAdmin(input: { email: string; password: string; displayName: string }) {
  return requireData(await api.POST('/api/auth/bootstrap', { body: input }));
}

export async function getMe() {
  return requireData(await api.GET('/api/auth/me'));
}

export async function login(input: { email: string; password: string }) {
  const result = await api.POST('/api/auth/login', { body: input });
  if (!result.response.ok) {
    requireData(result);
  }
}

export async function logout() {
  const result = await api.POST('/api/auth/logout');
  if (!result.response.ok) {
    requireData(result);
  }
}

export async function listSessions() {
  return requireData(await api.GET('/api/workbench/sessions'));
}

export async function getSession(sessionId: string) {
  return requireData(
    await api.GET('/api/workbench/sessions/{session_id}', {
      params: { path: { session_id: sessionId } },
    }),
  );
}

export async function listCandidateReviewItems(sessionId: string) {
  return requireData(
    await api.GET('/api/workbench/sessions/{session_id}/candidates', {
      params: { path: { session_id: sessionId } },
    }),
  );
}

export async function listGraphCandidates(sessionId: string, nodeId: string, cursor?: string, limit = 50) {
  return requireData(
    await api.GET('/api/workbench/sessions/{session_id}/graph-candidates', {
      params: {
        path: { session_id: sessionId },
        query: { node_id: nodeId, cursor, limit },
      },
    }),
  );
}

export async function getGraphCandidateResumeSnapshot(sessionId: string, graphCandidateId: string) {
  return requireData(
    await api.GET('/api/workbench/sessions/{session_id}/graph-candidates/{graph_candidate_id}/resume-snapshot', {
      params: {
        path: {
          session_id: sessionId,
          graph_candidate_id: graphCandidateId,
        },
      },
    }),
  );
}

export async function listSessionEvents(sessionId: string, afterSeq = 0) {
  const EVENT_PAGE_LIMIT = 200;
  const EVENT_MAX_PAGES = 25;
  const events: WorkbenchEvent[] = [];
  let cursor = afterSeq;

  for (let pageIndex = 0; pageIndex < EVENT_MAX_PAGES; pageIndex += 1) {
    const page = requireData(
      await api.GET('/api/workbench/sessions/{session_id}/events', {
        params: {
          path: { session_id: sessionId },
          query: { after_seq: cursor, limit: EVENT_PAGE_LIMIT },
        },
      }),
    );
    events.push(...page.events);
    if (page.events.length < EVENT_PAGE_LIMIT) {
      break;
    }
    cursor = page.events[page.events.length - 1]?.globalSeq ?? cursor;
  }

  return { events };
}
```

```svelte
<script lang="ts">
  import { untrack } from 'svelte';
  import { Background, Controls, SvelteFlow } from '@xyflow/svelte';
  import '@xyflow/svelte/dist/style.css';
  import type { Edge, NodeEventWithPointer, NodeTargetEventWithPointer } from '@xyflow/svelte';
  import type { RecruiterGraphNode } from '$lib/workbench/recruiterAnimation';
  import type { RunStory } from '$lib/workbench/runStory';
  import {
    fallbackLayout,
    layoutStrategyGraph,
    mergeManualNodePositions,
    type LaidOutStrategyGraph,
  } from '$lib/workbench/strategyGraphLayout';
  import StrategyGraphNode, { type StrategyFlowNode } from './StrategyGraphNode.svelte';

  type ManualPosition = { x: number; y: number };
  type StrategyFlowEdge = Edge;
  const defaultGraphBounds = { width: 980, height: 560 };
  const minGraphBounds = { width: 360, height: 420 };
  const nodeTypes = { strategy: StrategyGraphNode };

  let {
    story,
    selectedNodeId,
    onSelectNode,
  } = $props<{
    story: RunStory;
    selectedNodeId: string | null;
    onSelectNode: (node: RecruiterGraphNode) => void;
  }>();

  let shellElement = $state<HTMLDivElement>();
  let graphBounds = $state(defaultGraphBounds);
  let laidOutGraph = $state<LaidOutStrategyGraph>(fallbackLayout(story.graphNodes, story.graphEdges, graphBounds));
  let manualPositions = $state.raw(new Map<string, ManualPosition>());
  let previousGraphIdentity = $state<string | null>(null);
  let nodes = $state.raw<StrategyFlowNode[]>([]);
  let edges = $state.raw<StrategyFlowEdge[]>([]);

  const graphIdentity = $derived(story.graphNodes.map((node) => node.id).join('|'));

  function projectGraph(graph: LaidOutStrategyGraph, nextManualPositions = manualPositions) {
    nodes = graph.nodes.map((node) => {
      const selected = node.id === selectedNodeId;
      return {
        ...node,
        type: 'strategy',
        position: nextManualPositions.get(node.id) ?? node.position,
        selected,
        draggable: true,
        data: {
          ...node.data,
          selected,
          onSelectNode,
        },
      };
    }) as StrategyFlowNode[];
    edges = graph.edges as StrategyFlowEdge[];
  }

  $effect(() => {
    if (!shellElement) {
      return;
    }
    const updateBounds = () => {
      const rect = shellElement.getBoundingClientRect();
      graphBounds = {
        width: rect.width > 0 ? Math.max(minGraphBounds.width, Math.round(rect.width)) : defaultGraphBounds.width,
        height: rect.height > 0 ? Math.max(minGraphBounds.height, Math.round(rect.height)) : defaultGraphBounds.height,
      };
    };
    updateBounds();
    const observer = new ResizeObserver(updateBounds);
    observer.observe(shellElement);
    return () => observer.disconnect();
  });

  $effect(() => {
    let canceled = false;
    const fallback = fallbackLayout(story.graphNodes, story.graphEdges, graphBounds);
    laidOutGraph = fallback;

    void layoutStrategyGraph(story.graphNodes, story.graphEdges, graphBounds).then((graph) => {
      if (canceled) {
        return;
      }
      const manualPositionSnapshot = untrack(() => manualPositions);
      const currentGraphIdentity = untrack(() => previousGraphIdentity) ?? graphIdentity;
      const merged = mergeManualNodePositions({
        current: new Map(graph.nodes.map((node) => [node.id, node.position])),
        manual: manualPositionSnapshot,
        currentGraphIdentity,
        nextGraphIdentity: graphIdentity,
        nextNodeIds: graph.nodes.map((node) => node.id),
      });
      previousGraphIdentity = graphIdentity;
      manualPositions = merged.manualPositions;
      laidOutGraph = {
        ...graph,
        nodes: graph.nodes.map((node) => ({
          ...node,
          position: merged.positions.get(node.id) ?? node.position,
        })),
      };
    });
    return () => {
      canceled = true;
    };
  });

  $effect(() => {
    projectGraph(laidOutGraph, manualPositions);
  });

  const handleNodeClick: NodeEventWithPointer<MouseEvent | TouchEvent, StrategyFlowNode> = ({ node }) => {
    onSelectNode(node.data.graphNode);
  };

  const handleNodeDragStop: NodeTargetEventWithPointer<MouseEvent | TouchEvent, StrategyFlowNode> = ({ targetNode }) => {
    if (!targetNode) {
      return;
    }
    const next = new Map(manualPositions);
    next.set(targetNode.id, targetNode.position);
    manualPositions = next;
    projectGraph(laidOutGraph, next);
  };
</script>

<div class="strategy-flow-shell" data-testid="strategy-flow" bind:this={shellElement}>
  <SvelteFlow
    bind:nodes
    bind:edges
    {nodeTypes}
    minZoom={0.2}
    maxZoom={1.6}
    nodesDraggable={true}
    nodesConnectable={false}
    elementsSelectable={true}
    onnodeclick={handleNodeClick}
    onnodedragstop={handleNodeDragStop}
    fitView
  >
    <Background gap={24} size={1} />
    <Controls showInteractive={false} />
  </SvelteFlow>
</div>
```

Do not use React Flow event names or a `CustomEvent.detail` shape here. Svelte Flow click handlers receive typed callback arguments such as `{ node, event }`, while drag-stop handlers receive `{ targetNode, nodes, event }`; verify this against the installed `@xyflow/svelte` package by running `bun run check`.

Keep graph layout and graph projection separate. Selecting a node must only rebuild Svelte Flow node data; it must not trigger ELK layout again.

```svelte
    enabled: Boolean(selectedGraphCandidate?.canExpandResume && selectedGraphCandidate.graphCandidateId),
  }));

  const story = $derived(
    sessionQuery.data
      ? buildRunStory({
          session: sessionQuery.data,
          candidateReviewItems: candidatesQuery.data?.items ?? [],
          events: eventsQuery.data?.events ?? [],
        })
      : null,
  );
</script>

<section class="session-detail-page">
  {#if sessionQuery.isPending}
    <LoadingState />
  {:else if sessionQuery.error}
    <ErrorState message={safeErrorMessage(sessionQuery.error, '会话加载失败')} />
  {:else if story}
    <header class="job-brief">
      <h1>{sessionQuery.data.jobTitle}</h1>
      <p>{sessionQuery.data.notes}</p>
    </header>
    <div class="workbench-shell">
      <section class="graph-panel">
        <StrategyGraph
          {story}
          selectedNodeId={selectedNode?.id ?? null}
          onSelectNode={(node) => {
            selectedNode = node;
            selectedGraphCandidate = null;
          }}
        />
      </section>
      <NodeDetailPanel
        node={selectedNode}
        graphCandidates={graphCandidatesQuery.data?.items ?? []}
        graphCandidatesLoading={graphCandidatesQuery.isPending && Boolean(selectedNode)}
        graphCandidatesError={graphCandidatesQuery.error ? safeErrorMessage(graphCandidatesQuery.error, '候选人加载失败') : null}
        selectedGraphCandidateId={selectedGraphCandidate?.graphCandidateId ?? null}
        resumeSnapshot={resumeSnapshotQuery.data ?? null}
        resumeSnapshotLoading={resumeSnapshotQuery.isPending && Boolean(selectedGraphCandidate)}
        resumeSnapshotError={resumeSnapshotQuery.error ? safeErrorMessage(resumeSnapshotQuery.error, '简历摘要加载失败') : null}
        onSelectGraphCandidate={(candidate) => {
          selectedGraphCandidate = candidate;
        }}
      />
    </div>
  {/if}
</section>
```

If generated OpenAPI query parameter names differ from `after_seq` or `limit`, adjust only the typed API wrapper and record the generated names in task notes.
